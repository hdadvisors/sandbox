Data wrangle

proj_data <- estimate_raw |> 
  select(
    !geography) |> 
  mutate(
    fips = paste0("51", fips)) |> 
  left_join(
    proj_raw,
    by = "fips",
    keep = FALSE) |> 
  select(1, 4, 2, 3, 5:7)

proj_pct <- proj_data |> 
  select(2, 4, 7) |> 
  mutate(
    pct_change = (p2050 - e2021)/e2021,
    type = case_when(
      pct_change > 0 ~ "Risers",
      pct_change < 0 ~ "Fallers"
    ))

Top 10 plot

library(scales)
library(patchwork)

top <- proj_pct |> 
  slice_max(pct_change, n = 10) |> 
  ggplot(
    aes(x = pct_change,
        y = reorder(geography, pct_change))) +
    geom_col(fill = "#68718a") +
    geom_text(aes(label = label_percent(accuracy = 1)(pct_change)),
              color = "#011e41",
              nudge_x = -0.08) +
    scale_x_continuous(limits = c(0, 0.8)) +
    labs(subtitle = "Top 10") +
    theme(
      plot.margin = margin(0, 0, 0, 0, "pt"),
      panel.background = element_blank(),
      axis.title = element_blank(),
      axis.ticks = element_blank(),
      axis.text.x = element_blank(),
      plot.subtitle = element_text(hjust = 0.05,
                                   color = "#353535",
                                   margin = margin(0,0,0,0)))
